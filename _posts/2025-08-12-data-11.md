# 第2題  交錯字串 (Alternating Strings)
```python
def check(date):
    out=[]
    c=1
    for i in range(1,len(date)):
        if date[i].islower()==date[i-1].islower():
            c+=1

        else:
            out.append(c)
            c=1
    out.append(c)
    return out

def ans(k,date):
    c=0
    j=0
    for o in range(len(date)):
        if o+1==len(date):
            if date[o]>=k:
                c+=k
                if c>j:
                    j=c
                c=0
                
        elif date[o]>k:
            c+=k
            if date[o+1]>k:
                c+=k
                if c>j:
                    j=c
                c=0
            elif date[o+1]<k:
                if c>j:
                    j=c
                c=0
            
        elif date[o]==k:
            c+=k
            if date[o+1]>k:
                c+=k
                if c>j:
                    j=c
                c=0
            elif date[o+1]<k:
                if c>j:
                    j=c
                c=0
    return j
            
            
                


from sys import stdin
k=int(stdin.readline())
date=str(stdin.readline().strip())

m=check(date)
out2=ans(k,m)
print(out2)
print("")
```
