# 第3題 線段覆蓋長度 -小測資(前半題)
```markdown
N=int(input())
list3=[]
list2=[]
list4=[]
for i in range(N):
  list1=list(map(int,input().split()))
  list2.append(list1)
for t in range(1001):
  list3.append(t)
for y in list2:
  if y[0]==y[-1]:
    pass
  else:  
    for u in range(y[0]+1,y[-1]+1):
      try:
        list3.remove(u)
      except:
        pass
print(1001-len(list3))
```
