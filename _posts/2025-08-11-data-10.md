# 第1題 成績指標 
```
n=int(input())
list1=list(map(int,input().split()))
list1.sort()
list2=[]
list3=[]
print(*list1)
for i in list1:
  if i<60:
    list2.append(i)
  else:
    list3.append(i)
if len(list2)==0:
  print("best case")
else:
  print(list2[-1])
if len(list2)==len(list1):
  print("worst case")
else:
  print(list3[0])

```
