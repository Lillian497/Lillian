#  Zerojudgeu f605. 1. 購買力
```python
from sys import stdin
n,d=map(int,stdin.readline().split())
date=[]
c=0
an=[]
for i in range(n):
        list1=list(map(int,stdin.readline().split()))
        date.append(list1)
for o in date:
    if max(o)-min(o)>d:
        c+=1
        an.append(sum(o)//3)
print(c,sum(an))


```
