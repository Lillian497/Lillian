#  題目範例 第2題 小群體
```python
N=int(input())
list2=[0]*N
date=list(map(int,input().split()))


def find(y,u,c,list2):
          if y==u:
              c+=1
              list2[y]=1
              return c
          else:
              list2[y]=1
              return find(date[y],u,c,list2)
c=0
for w in range(N):
          if list2[w]==0:
              y=date[w]
              c=find(y,w,c,list2)
          elif sum(list2)==N:
              break
print(c)
print("")

```
